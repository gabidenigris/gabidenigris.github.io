/* Correção para o modo escuro - Gabriela De Nigris */ /* Força o esquema de cores a aceitar tanto light quanto dark */ html { color-scheme: light dark !important; } /* Quando o sistema ou o navegador estiver em modo escuro */ @media (prefers-color-scheme: dark) { body, .page__content, .initial-content { background-color: #1a1a1a !important; /* Fundo escuro */ color: #e0e0e0 !important; /* Texto claro */ } .page__title, .page__content h1, .page__content h2, .page__content h3, .page__content h4, .page__content p, .page__content li, .author__name, .author__bio, .blog-post-title, .blog-post-subtitle { color: #ffffff !important; /* Garante que títulos e textos fiquem brancos no modo escuro */ } .page__content a, .author__urls a, .blog-post-title a { color: #4da6ff !important; /* Links mais visíveis no fundo escuro */ } .greedy-nav .site-title, .greedy-nav a { color: #ffffff !important; } } /* Garante que no modo claro o texto continue visível se o fundo for claro */ @media (prefers-color-scheme: light) { body, .page__content, .initial-content { background-color: #fafaf8 !important; color: #111111 !important; } .page__title, .page__content h1, .page__content h2, .page__content h3, .page__content h4, .page__content p, .page__content li, .author__name { color: #111111 !important; } }
